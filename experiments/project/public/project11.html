Project based Experiment

About the experiment

User can add an album to favorite list. Profile page shows the favorite albums of the user and user can delete a favorite album from profile page.
Project link

Code Snippet

                Client Side
                $scope.addAlbumToFavorites = function (album) {
                    var currentUser = ($rootScope.currentUser);
                    var albumToBeUpdated = { name: album.name, id: album.id };
                    currentUser.albums.push(albumToBeUpdated);
                    $http.put("/api/user/" + currentUser._id, currentUser)
                    .success(function (response) {
                        $scope.users = response;
                        alert("Album \"" + album.name + "\" is added to favorites.");
                    });
                }

                $scope.removeAlbumFromFav = function (album) {
                    var currentUser = ($rootScope.currentUser);
                    var index = $scope.favAlbums.indexOf(album);
                    $scope.favAlbums.splice(index, 1);
                    currentUser.albums.splice(index, 1);
                    $http.put("/api/user/" + currentUser._id, currentUser)
                    .success(function (response) {
                        $scope.users = response;
                    });
                }

Code Snippet Explanation

Upon clicking the favorite button, album is added to the database and shown as a favorite album of the user. Favorite albums are shown on the profiles page and upon clicking the delete button, album is removed from the favorites.
