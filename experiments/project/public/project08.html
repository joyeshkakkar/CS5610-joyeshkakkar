About the experiment

User can click on fav button to add the track to his favorite list.

Code Snippet

                Client Side
                $scope.addTrackToFavorites = function (track) {
                    var currentUser = ($rootScope.currentUser);
                    var trackToBeUpdated = { name: track.name, id: track.id };
                    currentUser.tracks.push(trackToBeUpdated);
                    $http.put("/api/user/" + currentUser._id, currentUser)
                    .success(function (response) {
                        $scope.users = response;
                        alert("Track \""+ track.name +"\" is added to favorites.");
                    });
                }
                Server
                app.put("/api/user/:id", function (req, res) {
                    UserModel.update({ _id: req.params.id }, { $set: req.body }, function (err, doc) {
                        UserModel.find(function (err, data) {
                            res.json(data);
                        });
                    });
                });

Code Snippet Explanation

User marks a track as favorite which is added to database in users favorites list.
